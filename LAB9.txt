DECLARE 
   ROLL STUDENT.ROLL%type; 
   NAME STUDENT.NAME%type; 
   CODE COURSE.CODE%type; 
   CURSOR REPORTS is 
      SELECT ROLL,NAME FROM STUDENT;
      SELECT CODE FROM COURSE;
   counter int;

BEGIN 
   counter := 0;
   OPEN REPORTS; 
   LOOP 
   FETCH REPORTS into ROLL, NAME, CODE; 
   
      EXIT WHEN c_customers%notfound; 
      dbms_output.put_line(ROLL || ' ' || NAME || ' ' || CODE); 

      counter := counter+1;
   END LOOP; 
   CLOSE c_customers; 
END;
